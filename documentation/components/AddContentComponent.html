import {
  Component,
  computed,
  CUSTOM_ELEMENTS_SCHEMA,
  EventEmitter,
  output,
  Output,
  signal,
  WritableSignal,
} from '@angular/core';
import { FormsModule } from '@angular/forms';
import { eContentType } from 'src/_models/app.enums';
import { iMemorycardContent } from 'src/_models/domains/memorycard.models';

@Component({
  selector: 'app-add-content',
  imports: [FormsModule],
  schemas: [CUSTOM_ELEMENTS_SCHEMA],
  templateUrl: './add-content.component.html',
  styleUrl: './add-content.component.scss',
})
export class AddContentComponent {
  _CONTENT_TYPE_ENUM = eContentType;
  selContentType: WritableSignal<eContentType> = signal(
    this._CONTENT_TYPE_ENUM.text,
  );
  readonly memorycardTypes = Object.values(this._CONTENT_TYPE_ENUM); // Transforme en tableau pour itérer dessus

  destroyComponent = signal(false);

  returned_content = computed(() => ({
    value: '',
    mediaType: this.selContentType(), // Utilise la valeur actuelle de selContentType
    description: '',
  }));

  onChangeMedia(newContentType: eContentType) {
    console.log('NewContentType', newContentType);
    console.log('old selContentType', this.selContentType());
    this.selContentType.set(newContentType);
    console.log('new selContentType', this.selContentType());
  }
  previewUrl: string | ArrayBuffer | null = null;
  dataSent = output<iMemorycardContent>();

  isMediaSelected(media: eContentType): boolean {
    return media === this.selContentType();
  }
  async onValidate() {
    this.isContentValid();
    this.dataSent.emit(this.returned_content());
    const isDestroyed = await this.destroyComponent();

    if (isDestroyed === true) {
      this.ngOnDestroy();
    }
  }
  ngOnDestroy() {}
  isContentValid() {}
  dev() {
    console.log(this.returned_content());
  }

  imagePreview: string | null = null; // Contient l'aperçu de l'image
  selectedFile: File | null = null; // Contient le fichier sélectionné

  /**
   * Méthode appelée lorsque l'utilisateur sélectionne une image.
   */
  onImageSelected(event: Event): void {
    const input = event.target as HTMLInputElement;
    if (input.files && input.files.length > 0) {
      this.selectedFile = input.files[0];

      // Generate a preview of the selected photo
      const reader = new FileReader();
      reader.onload = () => {
        this.previewUrl = reader.result;
      };
      reader.readAsDataURL(this.selectedFile);
    }
  }

  /**
   * Enregistre l'image dans le localStorage.
   */
  saveImage(): void {
    if (this.imagePreview) {
      try {
        // Stocke l'image en Base64 dans le localStorage
        localStorage.setItem('savedImage', this.imagePreview);
        console.log('Image enregistrée dans le localStorage');
      } catch (error) {
        console.error("Erreur lors de la sauvegarde de l'image :", error);
      }
    }
  }

  /**
   * Charge une image précédemment enregistrée depuis le localStorage.
   */
  loadSavedImage(): void {
    const savedImage = localStorage.getItem('savedImage');
    if (savedImage) {
      this.imagePreview = savedImage;
      console.log('Image chargée depuis le localStorage');
    }
  }

  /**
   * Supprime l'image enregistrée du localStorage.
   */
  deleteSavedImage(): void {
    localStorage.removeItem('savedImage');
    this.imagePreview = null;
    console.log('Image supprimée du localStorage');
  }
}
